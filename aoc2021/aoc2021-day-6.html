function liveADay(school) {
    const countZeroes = school.filter(i => i === 0).length;
    return school
        .map(i => i < 1 ? 6 : i - 1)
        .concat(Array(countZeroes).fill(0).map(_ => 8));
}

function after80Days(school) {
    for (let i = 0; i < 80; i++) {
        school = liveADay(school);
        // console.log(`Lived through ${i}. School: ${school.length}`);
    }
    return school.length;
}

function liveADay(school) {
    const births = school.pop();
    school.unshift(births);
    school[2] += births;
    return school;
}

function after256Days(school) {
    let newSchool = [
        school.filter(i => i === 8).length || 0,
        school.filter(i => i === 7).length || 0,
        school.filter(i => i === 6).length || 0,
        school.filter(i => i === 5).length || 0,
        school.filter(i => i === 4).length || 0,
        school.filter(i => i === 3).length || 0,
        school.filter(i => i === 2).length || 0,
        school.filter(i => i === 1).length || 0,
        school.filter(i => i === 0).length || 0,
    ];
    // console.log(newSchool);
    for (let i = 0; i < 256; i++) {
        newSchool = liveADay(newSchool);
        // console.log(`Lived through ${i}. School: ${newSchool}, strength: ${newSchool.reduce((a, c) => a + c, 0)}`);
    }
    return newSchool.reduce((a, c) => a + c, 0);
}
