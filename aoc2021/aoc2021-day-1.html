JavaScript/nodejs

Part 1

    
#!/usr/bin/env node

/* Advent Of Code problem #1a */
const {readFileSync} = require('fs');
const input = readFileSync(0).toString();
const depths = input.split('\n').filter(i => i.length > 0).map(i => parseInt(i));
const increasing = depths.filter((v, i, a) => v < (a[i + 1] || -Infinity));
// const increasing = depths.filter((v, i, a) => (a[i + 1] || -Infinity) > v);
console.log(increasing.length);

#!/usr/bin/env node

/* Advent Of Code problem #1b */
const {readFileSync} = require('fs');
const input = readFileSync(0).toString();
const depths = input.split('\n').filter(i => i.length > 0).map(i => parseInt(i));
const windows = depths.map((v, i, a) => v + (a[i+1] || 0) + (a[i+2] || 0));
// const increasing = windows.filter((v, i, a) => v < (a[i + 1] || -Infinity));
const increasing = windows.filter((v, i, a) => (a[i + 1] || -Infinity) > v);
console.log(increasing.length);

#!/usr/bin/env python3

import sys
from more_itertools import windowed
from functools import reduce

lines = [line for line in sys.stdin]
numbers = list(map(lambda x: int(x), lines))
# print(numbers)
# print(reduce(lambda x,y: x+y, numbers))
window_size = 3
windows = list(windowed(numbers,n=3))
numbers = list(map(lambda x: reduce(lambda a,b: a+b, x), windows))
# print(windows, numbers)

increased = 0

for i, v in enumerate(numbers):
    if i < len(numbers)-1:
        if v < numbers[i+1]:
            increased += 1
