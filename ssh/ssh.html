ssh

~/.ssh/config file 

    Host github.com
        User YOUR_GIT_USER_NAME
        Hostname github.com
        PreferredAuthentications publickey
        IdentitiesOnly yes
        AddKeysToAgent yes
        IdentityFile ~/.ssh/YOUR_PRIVATE_KEY
        # UseKeychain yes # mac only

ssh-add

    When run without arguments, it adds the files
    ~/.ssh/id_rsa, ~/.ssh/id_ecdsa, ~/.ssh/id_ecdsa_sk,
    ~/.ssh/id_ed25519,  ~/.ssh/id_ed25519_sk, and ~/.ssh/id_dsa.

ssh-keygen

    ssh-keygen -t ed25519 -b 4096 -C "ssh-key for ..."

Adding keys to ssh-agent

    eval "$(ssh-agent -s)"
    ssh-add -t 1000000 # OR
    ssh-add -t 1000000 ~/.ssh/KEY_NAME

List all keys in the ssh-agent

    ssh-add -l
